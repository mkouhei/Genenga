#!/bin/sh

(
cd $(git rev-parse --show-toplevel)
find genenga -name '*.pyc' -delete
python setup.py test || exit 1
python setup.py check -r || exit 1
python3 setup.py check -r || exit 1
pychecker genenga/*.py
pychecker -q genenga/tests/*.py
pylint -d too-few-public-methods \
    -d too-many-public-methods \
    -d superfluous-parens \
    -d no-self-use \
    genenga
)
